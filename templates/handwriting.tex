    % you may add more values here and then use them in the {tikzpicture}

% package to have a "every line" hook
\usepackage{lineno}

\newenvironment{drillsheet}{%
    % let the "line numbers" start directly at the text
    \setlength\linenumbersep{0pt}%
    % let the "line numbers" be the grid
    \renewcommand{\thelinenumber}{%
        \begin{tikzpicture}[overlay]
                \color{black!90}%
                % baseline
                \draw [semithick] ((0,0) -- ++(\textwidth,0);
                % lowercase height
                \draw [semithick, densely dotted] ((0,\lowercaseheight) -- ++(\textwidth,0);
                % uppercase height
                \draw [semithick] ((0,\uppercaseheight) -- ++(\textwidth,0);
                % descender depth
                \draw [semithick] (0,\descenderdeptha) -- ++(\textwidth,0);
                \draw [semithick, densely dotted] (0,\descenderdepthb) -- ++(\textwidth,0);
                \draw [semithick] (0,\descenderdepthc) -- ++(\textwidth,0);
        \end{tikzpicture}%
    }%
    % the font and color of the numbers must be set explicitly
    \renewcommand\linenumberfont{\normalfont\color{black}}
    % start "line numbering", i.e. the grid
    \begin{linenumbers}%
        % set text color to gray
        \color{black}%
}{
    % end "numbering", i.e. the grid
    \end{linenumbers}%
}
